/Unions/{unionId}
    - name: String
    - level: String
    - parentUnionId: String (optional)
    - contactPerson: String
    - contactPhone: String

/Kijiwes/{kijiweId}
    - location: GeoPoint
    - adminId: String (userId from Users collection)
    - createdAt: timestamp
    - name: String
    - permanentMembers: Array of Strings (userIds from Users collection)
    - unionId: String
    - position: Map  // For GeoFlutterFirePlus
        - geohash: String
        - geoPoint: GeoPoint
    - queue: Array of maps
        - driverId: String
        - joinedAt: Timestamp

/Users/{userId}  //  Unified User Collection
    - customerId: String (userId from Users collection)
    - createdAt: Timestamp
    - dob: String
    - driverProfile: Map  //  Optional - Only present if role includes "Driver"
        - approved: Boolean  // Whether they've been approved as a driver by an admin
        - isOnline: Boolean  // Whether they are currently available for rides and should be in the Kijiwe queue
        - kijiweId: String
        - licenseNumber: String
        - registeredAt: Timestamp
        - vehicleType: String
        - currentLocation: GeoPoint (Driver's active location)
        - rating: Number
        - status: String ("offline", "waitingForRide", "goingToPickup", "arrivedAtPickup", "onRide", "returningToKijiwe") // Example statuses
        - completedRides: Number // Counter for completed rides    - email: String
    - emailVerified: Boolean
    - fcmToken: String
    - gender: String
    - location: GeoPoint (General user location - can be used before driver status)
    - name: String
    - phoneNumber: String
    - profileImageUrl: String
    - role: String  // "Customer", "Driver", "Both", "Kijiwe Leader", "Union Official"
    - uid: String  // Firebase Auth UID (This is the key identifier)
    - photoURL: String
    

/RideRequests/{rideRequestId}
    - customerId: String (userId from Users collection)
    - assignedDriverId: String (userId from Users collection)
    - dropOff: GeoPoint
    - pickup: GeoPoint
    - status: String
    - stops: Array of GeoPoints
    - requestTime: Timestamp
    - acceptedTime: Timestamp
    - completedTime: Timestamp
    - route: Array of GeoPoints
    - kijiweId: String
    - fare: Number
